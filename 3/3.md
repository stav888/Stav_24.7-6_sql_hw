```sql
-- ON DELETE CASCADE

CREATE TABLE directors (
    director_id INTEGER PRIMARY KEY,
    name TEXT
);
```

```sql
CREATE TABLE movies (
    movie_id INTEGER PRIMARY KEY,
    title TEXT,
    director_id INTEGER,
    FOREIGN KEY(director_id) REFERENCES directors(director_id)
);
```
